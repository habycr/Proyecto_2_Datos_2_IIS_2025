===========================
1) palindrome-number
===========================

--- WRONG ANSWER ---
#include <iostream>
using namespace std;
int main() {
    long long x; 
    cin >> x;
    cout << "false";
}

--- TLE ---
#include <iostream>
using namespace std;
int main() {
    long long x;
    cin >> x;
    while (true) {}
}

--- MLE ---
#include <iostream>
#include <vector>
using namespace std;
int main() {
    long long x; 
    cin >> x;
    vector<int> v(2000000000, 7);
    cout << "false";
}
